# Step 9: Pipeline with scaling + polynomial transformation + linear regression
pipe = Pipeline([
    ("scale", StandardScaler()),
    ("poly", PolynomialFeatures(degree=2, include_bias=False)),
    ("model", LinearRegression())
])
pipe.fit(X_multi, y)
print("R^2 with polynomial pipeline:", pipe.score(X_multi, y))

R^2 with polynomial pipeline: 0.7513467656307783
